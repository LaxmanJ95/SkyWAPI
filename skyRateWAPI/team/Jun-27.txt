use skyrate;
CREATE TABLE `conversation_mapping` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `MESSENGER1` int(11) NOT NULL,
  `MESSENGER2` int(11) NOT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8;

CREATE TABLE `messenger` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `FROM_ID` int(11) NOT NULL,
  `TO_ID` int(11) NOT NULL,
  `CONVERSATION_ID` varchar(100) COLLATE utf8_bin NOT NULL,
  `REVIEW_ID` int(11) DEFAULT NULL,
  `MESSAGE` longtext COLLATE utf8_bin NOT NULL,
  `DATE_TIME` datetime DEFAULT NULL,
  `READ_FLAG` int(1) DEFAULT '0',
  PRIMARY KEY (`ID`)
) ENGINE=MyISAM AUTO_INCREMENT=273 DEFAULT CHARSET=utf8 COLLATE=utf8_bin;
